async function calculateJob() { const apiKey = document.getElementById('apiKey').value.trim(); const make = document.getElementById('vehicleMake').value; const model = document.getElementById('vehicleModel').value; const job = document.getElementById('jobInput').value.trim(); if (!apiKey || !make || !model || !job) { alert("Make sure to fill in all the blanks, Mr. Rivera!"); return; } // Show status document.getElementById('loading').style.display = 'block'; try { const response = await fetch('https://api.groq.com/openai/v1/chat/completions', { method: 'POST', headers: { 'Authorization': `Bearer ${apiKey}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ model: "llama3-8b-8192", messages: [ { role: "system", content: "You are a mechanic. Output ONLY: BOOK_HOURS: [number], REQUIRED_TOOLS: [text], TRAPS_AND_TRICKS: [text]" }, { role: "user", content: `Quote: ${make} ${model} ${job}` } ], temperature: 0.1 }) }); const data = await response.json(); // Check if we actually got a response if (data.choices && data.choices[0].message.content) { console.log("AI Response:", data.choices[0].message.content); alert("Connection successful! Check your console for the data."); } else { throw new Error("Empty response from AI"); } } catch (err) { console.error("The error is:", err); alert("Still hitting a wall. Check the console (Right-Click -> Inspect -> Console)."); } finally { document.getElementById('loading').style.display = 'none'; } }
